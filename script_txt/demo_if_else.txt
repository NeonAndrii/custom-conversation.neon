Script: "Demo If Else"
Variable: {conditional}: "continue"
Variable: {test}: "False"

if {test}:
    Neon speak: "True"
else:
    Neon speak: "False"

Neon speak: "Say {conditional} to continue"
voice_input{test}

if {test} != {conditional}:
    Neon speak: "Not matched, exiting"
    Exit
else:
    Neon speak: "Okay, lets continue."
Neon speak: "Outside of if statement"
Exit
